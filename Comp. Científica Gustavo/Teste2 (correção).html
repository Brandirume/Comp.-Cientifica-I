Neste teste vamos estudar a evoluo do erro no eixo $y$ no decorrer do algoritmo da bisseo. Para isso, vamos usar uma variao do mtodo da bisseo que retorne uma lista com todos os pontos percorridos pelo algoritmo at atingir o critrio de parada.

def bissecao_pts(f, a, b, xtol=1e-8, ytol=1e-8, maxiter=100):
    pts=[]
    m = (a+b)/2
    fm = f(m)
    iters=1
    pts.append(m)
    while (abs(b-a) >= xtol) and (iters < maxiter) and (abs(fm) > ytol):
        if fm*f(a) < 0:
            b = m
            m = (a+b)/2
            fm = f(m)
            if a not in pts:
                pts.append(a)
            if m not in pts:
                pts.append(m)
        else:
            a = m
            m = (a+b)/2
            fm = f(m)
            if m not in pts:
                pts.append(m)
        iters+=1
    return pts
